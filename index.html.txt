        // Simulasi model CNN (menggantikan model.predict dari Python)
        function predictImage(imageData) {
            // Simulasi prediksi berdasarkan analisis pixel (dalam implementasi nyata, 
            // kirim ke Flask/Django backend dengan TensorFlow.js atau API)
            const canvas = document.createElement('canvas');
            const ctx = canvas.getContext('2d');
            canvas.width = 32;
            canvas.height = 32;
            
            // Resize dan preprocess image
            ctx.drawImage(imageData, 0, 0, 32, 32);
            const imageDataResized = ctx.getImageData(0, 0, 32, 32);
            
            // Analisis sederhana berdasarkan warna kuning (R,G,B)
            let yellowPixels = 0;
            const data = imageDataResized.data;
            for (let i = 0; i < data.length; i += 4) {
                const r = data[i];
                const g = data[i + 1];
                const b = data[i + 2];
                // Deteksi warna kuning (R tinggi, G sedang, B rendah)
                if (r > 180 && g > 120 && b < 100 && r > g) {
                    yellowPixels++;
                }
            }
            
            const totalPixels = (32 * 32);
            const yellowRatio = yellowPixels / totalPixels;
            
            // Klasifikasi berdasarkan rasio kuning
            let prediction, confidence, status;
            if (yellowRatio < 0.1) {
                prediction = 0.05; // Sehat
                confidence = 0.95;
                status = 'status-sehat';
            } else if (yellowRatio < 0.4) {
                prediction = 0.65; // Ringan
                confidence = 0.85;
                status = 'status-ringan';
            } else {
                prediction = 0.92; // Parah
                confidence = 0.92;
                status = 'status-parah';
            }
            
            return {
                prediction: [prediction],
                confidence: confidence,
                yellowRatio: yellowRatio,
                status: status
            };
        }

            uploadSection.classList.remove('dragover');
            const files = e.dataTransfer.files;
            if (files.length > 0) {
                handleImageUpload(files[0]);
            }
        });

        uploadArea.addEventListener('click', () => {
            uploadImage.click();
        });

        uploadImage.addEventListener('change', (e) => {
            if (e.target.files.length > 0) {
                handleImageUpload(e.target.files[0]);
            }
        });

        function handleImageUpload(file) {
            // Validasi file
            if (!file.type.startsWith('image/')) {
                showError('Harap pilih file gambar (JPG, PNG, dll.)');
                return;
            }

            if (file.size > 5 * 1024 * 1024) {
                showError('Ukuran file maksimal 5MB');
                return;
            }

            const reader = new FileReader();
            reader.onload = function(e) {
                const img = new Image();
                img.onload = function() {
                    processImage(img);
                };
                img.src = e.target.result;
            };
            reader.readAsDataURL(file);
        }

        function processImage(img) {
            // Tampilkan loading
            loading.style.display = 'block';
            resultImageContainer.style.display = 'none';
            predictionResult.classList.remove('show');
            hideError();

            // Simulasi delay pemrosesan model (2-3 detik)
            setTimeout(() => {
                // Tampilkan gambar hasil
                resultImage.src = img.src;
                resultImageContainer.style.display = 'block';

                // Prediksi menggunakan model simulasi
                const result = predictImage(img);
                
                // Tampilkan hasil
                showPrediction(result);
                loading.style.display = 'none';
            }, 2500);
        }

        function showPrediction(result) {
            const labels = ['Daun Sehat', 'Kuning Ringan', 'Kuning Parah'];
            const statusLabels = ['Daun Sehat (Healthy)', 'Kuning Ringan (Mild)', 'Kuning Parah (Severe)'];
            
            const predictedClass = result.prediction[0] <= 0.33 ? 0 : 
                                 result.prediction[0] <= 0.66 ? 1 : 2;
            
            predictionTitle.textContent = `Status: ${statusLabels[predictedClass]}`;
            confidenceScore.textContent = `${(result.confidence * 100).toFixed(1)}%`;
            healthStatus.textContent = statusLabels[predictedClass];
            healthStatus.className = `health-status ${result.status}`;
            
            predictionResult.classList.add('show');
        }

        function showError(message) {
            errorMessage.textContent = message;
            errorMessage.style.display = 'block';
            loading.style.display = 'none';
        }

        function hideError() {
            errorMessage.style.display = 'none';
        }
